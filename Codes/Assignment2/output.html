def plotvariable(df, variable):
  plt.figure(figsize=(16,4))
  #histogram
  plt.subplot(1,2,1)
  plt.hist(df[variable],alpha = .5)

  #boxplot
  plt.subplot(1,2,2)
  plt.boxplot(df[variable])
  plt.show()
